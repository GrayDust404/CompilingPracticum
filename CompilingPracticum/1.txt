program   Exam52(input,output);
const   N=10;
var a: array[1..10] of integer;i,j: integer;
procedure swap(var x,y: integer);              
var t:integer;
begin  t:=x;   x:=y;   y:=t end;
begin
    for i:=1 to N do                                       
      begin   
	write(i);
      end;
    for j:=1 to N do                                
      for i:=1 to N-j do                              
         if a[ i ] < a[i+1] then swap(a[ i ], a[i+1]);    
    for i:=1 to N do                                   
      write(a[ i ]);
end.
